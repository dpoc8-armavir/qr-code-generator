    function generateQRCode(type) {
      let value = "";
      let containerId = "";

      if (type === "text") {
        value = document.getElementById("text").value.trim();
        containerId = "qr-text";
      } else if (type === "link") {
        value = document.getElementById("link").value.trim();
        containerId = "qr-link";
      } else if (type === "image") {
        const file = document.getElementById("image").files[0];
        containerId = "qr-image";

        if (!file) {
          alert("‘Ω’∂’§÷Ä’∏÷Ç’¥ ’•’∂÷Ñ ’®’∂’ø÷Ä’•’¨ ’∂’Ø’°÷Ä÷â");
          return;
        }

        const reader = new FileReader();
        reader.onload = function(e) {
          const imgData = e.target.result; // base64 ’∂’Ø’°÷Ä
          createQR(imgData, containerId);
        };
        reader.readAsDataURL(file);
        return;
      }

      if (value === "") {
        alert("‘Ω’∂’§÷Ä’∏÷Ç’¥ ’•’∂÷Ñ ’¨÷Ä’°÷Å’∂’•’¨ ’§’°’∑’ø’®÷â");
        return;

      createQR(value, containerId);
    }

      };

      container.appendChild(img);
